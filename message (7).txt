local function disconnectEvent(pattern)
    for key, connection in eventConnections do
        if key:find(pattern) then
            connection:Disconnect()
            connection = nil
        end
    end
end

local function getPlayerPlotItem()
    for _, plot in pairs(workspaceService.Plots:GetChildren()) do
        local plotSign = plot:FindFirstChild('PlotSign')
        local ownerList = plotSign and plotSign:FindFirstChild('ThisPlotsOwners')
        if ownerList then
            for _, owner in pairs(ownerList:GetChildren()) do
                if (owner.Value == localPlayer.Name) then
                    local plotItem = workspaceService.PlotItems:FindFirstChild(plot.Name)
                    if plotItem then
                        return plotItem
                    end
                end
            end
        end
    end
    return nil
end

local function spawnToy(toyName, position, rotation)
    local inPlotFlag = localPlayer.InPlot
    local inOwnedPlotFlag = localPlayer.InOwnedPlot
    local canSpawnFlag = localPlayer.CanSpawnToy

    while inPlotFlag.Value and not inOwnedPlotFlag.Value and not canSpawnFlag.Value do
        task.wait()
    end

    if (typeof(rotation) == 'string') then
        if (string.lower(rotation) == 'headup') then
            task.spawn(function()
                spawnToyFunction:InvokeServer(toyName, CFrame.new(position.Position + Vector3.new(0, 1, 0), position.Position) * CFrame.Angles(math.pi, 0, 0), Vector3.new(0, 0, 0))
            end)
        end
    elseif rotation then
        task.spawn(function()
            spawnToyFunction:InvokeServer(toyName, position, rotation)
        end)
    else
        task.spawn(function()
            spawnToyFunction:InvokeServer(toyName, position, Vector3.new(0, 0, 0))
        end)
    end

    local spawnedModel, isInPlot

    if inOwnedPlotFlag.Value then
        local plotItem = getPlayerPlotItem()
        if plotItem then
            plotItem.ChildAdded:Once(function(child)
                if ((child.Name == toyName) and child:IsA('Model')) then
                    spawnedModel = child
                end
            end)
            isInPlot = true
        end
    else
        playerSpawnFolder.ChildAdded:Once(function(child)
            if ((child.Name == toyName) and child:IsA('Model')) then
                spawnedModel = child
            end
        end)
        isInPlot = false
    end

    local startTime = tick()
    while not spawnedModel do
        if ((tick() - startTime) < 3) then
            task.wait()
        else
            return false
        end
    end
    return spawnedModel, isInPlot
end

local function initBlobRide()
    local runService = game:GetService('RunService')
    local character = localPlayer.Character or localPlayer.CharacterAdded:Wait()
    local humanoid = character:WaitForChild('Humanoid')
    local rootPart = character:WaitForChild('HumanoidRootPart')
    trainConnection = rootPart.Position
    local spawnFolder = workspace:FindFirstChild(localPlayer.Name .. 'SpawnedInToys')
    local blobModel = spawnFolder and spawnFolder:FindFirstChild('CreatureBlobman')
    local vehicleSeat = blobModel and blobModel:FindFirstChild('VehicleSeat')
    if not blobModel then
        spawnBlobMan()
        task.wait(1)
        spawnFolder = workspace:FindFirstChild(localPlayer.Name .. 'SpawnedInToys')
        blobModel = spawnFolder and spawnFolder:FindFirstChild('CreatureBlobman')
        vehicleSeat = blobModel and blobModel:FindFirstChild('VehicleSeat')
    end
    if (vehicleSeat and vehicleSeat:IsA('VehicleSeat')) then
        rootPart.CFrame = vehicleSeat.CFrame + Vector3.new(0, 2, 0)
        vehicleSeat:Sit(humanoid)
    end
    humanoid:GetPropertyChangedSignal('Jump'):Connect(function()
        if (humanoid.Jump and humanoid.Sit) then
            blobJumpCounter = 15
            trainConnection = rootPart.Position
        end
    end)
    if blobConnection then
        blobConnection:Disconnect()
    end
    blobConnection = runService.Heartbeat:Connect(function()
        if (not rootPart or not humanoid) then
            return
        end
        replicatedStorage.CharacterEvents.RagdollRemote:FireServer(rootPart, 0)
        if (blobJumpCounter > 0) then
            rootPart.CFrame = CFrame.new(trainConnection)
            blobJumpCounter = blobJumpCounter - 1
        end
    end)
    task.spawn(function()
        while humanoid.Sit do
            task.wait(1)
        end
        task.wait(0.5)
        rootPart.CFrame = CFrame.new(trainConnection)
    end)
end

    end
end

                        end

local trainLoopConnection

local function initTrainRide()
    local runService = game:GetService('RunService')
    local character = localPlayer.Character or localPlayer.CharacterAdded:Wait()
    local humanoid = character:WaitForChild('Humanoid')
    local rootPart = character:WaitForChild('HumanoidRootPart')
    trainRootPos = rootPart.Position
    local trainObject = workspace.Map.AlwaysHereTweenedObjects
    local train = trainObject and trainObject:FindFirstChild('Train')
    local trainSeat
    if train then
        for _, descendant in ipairs(train:GetDescendants()) do
            if descendant:IsA('Seat') then
                trainSeat = descendant
                break
            end
        end
    end
    if trainSeat then
        rootPart.CFrame = trainSeat.CFrame + Vector3.new(0, 2, 0)
        trainSeat:Sit(humanoid)
    end
    humanoid:GetPropertyChangedSignal('Jump'):Connect(function()
        if (humanoid.Jump and humanoid.Sit) then
            trainJumpCounter = 15
            trainRootPos = rootPart.Position
        end
    end)
    if trainLoopConnection then
        trainLoopConnection:Disconnect()
    end
    trainLoopConnection = runService.Heartbeat:Connect(function()
        if (not rootPart or not humanoid) then
            return
        end
        replicatedStorage.CharacterEvents.RagdollRemote:FireServer(rootPart, 0)
        if (trainJumpCounter > 0) then
            rootPart.CFrame = CFrame.new(trainRootPos)
            trainJumpCounter = trainJumpCounter - 1
        end
    end)
    task.spawn(function()
        while humanoid.Sit do
            task.wait(1)
        end
        task.wait(0.5)
        rootPart.CFrame = CFrame.new(trainRootPos)
    end)
end

local function stopTrainRide()
    if trainLoopConnection then
        trainLoopConnection:Disconnect()
        trainLoopConnection = nil
    end
end

local function setupAntiLag()
    local runService = game:GetService('RunService')
    local repStorage = Game:GetService('ReplicatedStorage')
    local currentCharacter = localPlayer.Character or localPlayer.CharacterAdded:Wait()
    localPlayer.CharacterAdded:Connect(function(character)
        currentCharacter = character
    end)
    local isAntiLagActive = false
    local heartbeatConnection
    local function disableCollision()
        if not currentCharacter then
            return
        end
        for _, part in ipairs(currentCharacter:GetDescendants()) do
            if part:IsA('BasePart') then
                part.CanTouch = false
                part.CanCollide = false
                part.Massless = true
            end
        end
    end
    local function startAntiLagLoop()
        if heartbeatConnection then
            heartbeatConnection:Disconnect()
        end
        heartbeatConnection = runService.Heartbeat:Connect(function()
            if isAntiLagActive then
                disableCollision()
            end
        end)
    end
    local function stopAntiLagLoop()
        isAntiLagActive = false
        if heartbeatConnection then
            heartbeatConnection:Disconnect()
            heartbeatConnection = nil
        end
    end
    local permRagdollActive = nil
    local isPermRagdollLooping = false
    local isAntiRagdollActive = false
    local isRagdollLoopRunning = false
    local antiRagdollTarget = nil
    local shouldPermRagdoll = false
    local autoGucciActive = false
    local blobmanModel = nil
    function setRagdoll(enable)
        local character = localPlayer.Character
        local rootPart = character:WaitForChild('HumanoidRootPart')
        local humanoid = character:WaitForChild('Humanoid')
        if (character and character:FindFirstChild('HumanoidRootPart')) then
            repStorage:WaitForChild('CharacterEvents'):WaitForChild('RagdollRemote'):FireServer(rootPart, (enable and 1) or 0)
            if humanoid then
                humanoid.PlatformStand = enable
            end
        end
    end
    function permRagdollLoop()
        if isPermRagdollLooping then
            return
        end
        isPermRagdollLooping = true
        while shouldPermRagdoll do
            setRagdoll(true)
            task.wait(0.1)
        end
        isPermRagdollLooping = false
        setRagdoll(false)
    end
    function getBlobman()
        updatePlotNumber()
        if (currentPlotNum == 0) then
            if inv then
                return inv:FindFirstChild('CreatureBlobman')
            end
            return nil
        else
            return workspace.PlotItems:FindFirstChild('Plot' .. currentPlotNum):FindFirstChild('CreatureBlobman')
        end
    end
    function spawnBlobman()
        local character = localPlayer.Character
        local rootPart = character:WaitForChild('HumanoidRootPart')
        local spawnRemote = repStorage:WaitForChild('MenuToys'):WaitForChild('SpawnToyRemoteFunction')
        if spawnRemote then
            pcall(function()
                spawnRemote:InvokeServer('CreatureBlobman', rootPart.CFrame * CFrame.new(0, 0, -5), Vector3.new(0, -15.716, 0))
            end)
            task.wait(0.25)
            blobmanModel = getBlobman()
        end
    end
    function ragdollLoop()
        local character = localPlayer.Character
        local rootPart = character:WaitForChild('HumanoidRootPart')
        if isRagdollLoopRunning then
            return
        end
        isRagdollLoopRunning = true
        while isAntiRagdollActive do
            if (character and rootPart) then
                local params = {
                    [1] = rootPart,
                    [2] = 0,
                }
                repStorage:WaitForChild('CharacterEvents'):WaitForChild('RagdollRemote'):FireServer(unpack(params))
            end
            task.wait()
        end
        isRagdollLoopRunning = false
    end
    function sitAndJumpLoop()
        local character = localPlayer.Character
        if not character then
            return
        end
        local humanoid = character:FindFirstChildOfClass('Humanoid')
        local rootPart = character:FindFirstChild('HumanoidRootPart')
        if (not humanoid or not rootPart) then
            return
        end
        local startTime = tick()
        local hasJumped = false
        while autoGucciActive and ((tick() - startTime) < 6) do
            if blobmanModel then
                local seat = blobmanModel:FindFirstChildOfClass('VehicleSeat')
                if (seat and (seat.Occupant ~= humanoid)) then
                    seat:Sit(humanoid)
                end
            end
            task.wait(0.3)
            if not hasJumped then
                humanoid:ChangeState(Enum.HumanoidStateType.Jumping)
                hasJumped = true
            end
            local seat = blobmanModel:FindFirstChildOfClass('VehicleSeat')
            local bodyPos = Instance.new('BodyPosition')
            bodyPos.MaxForce = Vector3.new(math.huge, math.huge, math.huge)
            bodyPos.Position = seat.Parent:WaitForChild('HumanoidRootPart').Position + Vector3.new(0, 1000000, 100000)
            bodyPos.D = 10000000
            bodyPos.P = 100000000000
            bodyPos.Parent = seat.Parent:WaitForChild('HumanoidRootPart')
        end
        autoGucciActive = false
        isAntiRagdollActive = false
    end
    function updatePlotNumber()
        local character = localPlayer.Character
        if (character.Parent == workspace) then
            currentPlotNum = 0
        elseif (character.Parent.Name == 'PlayersInPlots') then
            for _, plot in workspace.Plots:GetChildren() do
                for _, owner in plot.PlotSign.ThisPlotsOwners:GetChildren() do
                    if (owner.Value == localPlayer.Name) then
                        if (owner.Parent.Parent.Parent.Name == 'Plot1') then
                            currentPlotNum = 1
                        elseif (owner.Parent.Parent.Parent.Name == 'Plot2') then
                            currentPlotNum = 2
                        elseif (owner.Parent.Parent.Parent.Name == 'Plot3') then
                            currentPlotNum = 3
                        elseif (owner.Parent.Parent.Parent.Name == 'Plot4') then
                            currentPlotNum = 4
                        elseif (owner.Parent.Parent.Parent.Name == 'Plot5') then
                            currentPlotNum = 5
                        end
                    end
                end
            end
        end
    end
    invincibilityTab:AddToggle({
        Name = 'Auto Gucci (manutencion)',
        Default = false,
        Callback = function(state)
            autoGucciActive = state
            if autoGucciActive then
                spawnBlobman()
                task.wait()
                if not isAntiRagdollActive then
                    coroutine.wrap(sitAndJumpLoop)()
                    isAntiRagdollActive = true
                end
                coroutine.wrap(ragdollLoop)()
            else
                isAntiRagdollActive = false
            end
        end,
    })

        Name = 'Ativar SkyBlobman',
        Callback = function()
            local function findBlobman()
                local spawnFolder = workspace:FindFirstChild(localPlayer.Name .. 'SpawnedInToys')
                if (spawnFolder and spawnFolder:FindFirstChild('CreatureBlobman')) then
                    return spawnFolder.CreatureBlobman
                end
                if workspace:FindFirstChild('PlotItems') then
                    for _, plotItem in pairs(workspace.PlotItems:GetChildren()) do
                        local blob = plotItem:FindFirstChild('CreatureBlobman')
                        if blob then
                            return blob
                        end
                    end
                end
                return nil
            end
            local function launchBlobmanToSky()
                local blob = findBlobman()
                if not blob then
                    return
                end
                local rootPart = blob:FindFirstChild('HumanoidRootPart')
                if not rootPart then
                    return
                end
                local bodyPos = Instance.new('BodyPosition')
                bodyPos.MaxForce = Vector3.new(math.huge, math.huge, math.huge)
                bodyPos.D = 10000000
                bodyPos.P = 100000000000
                bodyPos.Parent = rootPart
                bodyPos.Position = rootPart.Position + Vector3.new(0, 1000000, 100000)
            end
            launchBlobmanToSky()
        end,
    })
end

grabTab:AddToggle({

    Default = false,
    Callback = function(state)
        _G.stumpGrabEnabled = state
        if state then
            stumpConnection = Workspace.ChildAdded:Connect(function(child)
                if not _G.stumpGrabEnabled then
                    return
                end
                if (child.Name ~= 'GrabParts') then
                    return
                end
                task.wait()
                local grabPart = child:FindFirstChild('GrabPart')
                if not grabPart then
                    return
                end
                local weldConstraint = grabPart:FindFirstChildWhichIsA('WeldConstraint')
                if (not weldConstraint or not weldConstraint.Part1) then
                    return
                end
                local targetModel = weldConstraint.Part1.Parent
                if not targetModel then
                    return
                end
                local humanoid = targetModel:FindFirstChild('Humanoid')
                local rootPart = targetModel:FindFirstChild('HumanoidRootPart')
                local isDestroyed = false
                child.AncestryChanged:Connect(function(_, parent)
                    if not parent then
                        isDestroyed = true
                    end
                end)
                if (not humanoid or not rootPart) then
                    return
                end
                local function shouldStop()
                    return isDestroyed or not _G.stumpGrabEnabled
                end
                local head = targetModel:WaitForChild('Head', 3)
                if (not head or shouldStop()) then
                    return
                end
                local torso = targetModel:FindFirstChild('Torso')
                if not torso then
                    return
                end
                local torsoCFrame = torso.CFrame
                local headCFrame = head.CFrame
                local rootCFrame = rootPart.CFrame
                if shouldStop() then
                    return
                end
                local neckConstraint = head:WaitForChild('BallSocketConstraint', 3)
                if (not neckConstraint or shouldStop()) then
                    return
                end
                repeat
                    task.wait()
                    if shouldStop() then
                        return
                    end
                until (neckConstraint.Enabled == true) or (neckConstraint.Active == true)
                rootPart.Velocity = Vector3.zero
                local limbsToRemove = {}
                if _G.stumpLimbsToRemove.LeftLeg then
                    local leftLeg = targetModel:FindFirstChild('Left Leg')
                    if leftLeg then
                        table.insert(limbsToRemove, leftLeg)
                    end
                end
                if _G.stumpLimbsToRemove.RightLeg then
                    local rightLeg = targetModel:FindFirstChild('Right Leg')
                    if rightLeg then
                        table.insert(limbsToRemove, rightLeg)
                    end
                end
                if _G.stumpLimbsToRemove.LeftArm then
                    local leftArm = targetModel:FindFirstChild('Left Arm')
                    if leftArm then
                        table.insert(limbsToRemove, leftArm)
                    end
                end
                if _G.stumpLimbsToRemove.RightArm then
                    local rightArm = targetModel:FindFirstChild('Right Arm')
                    if rightArm then
                        table.insert(limbsToRemove, rightArm)
                    end
                end
                local function repositionLimbs(xPos, yPos)
                    if shouldStop() then
                        return
                    end
                    for _, limb in ipairs(limbsToRemove) do
                        if limb then
                            limb.CFrame = CFrame.new(10000, yPos, 10000)
                        end
                    end
                    rootPart.Velocity = Vector3.zero
                    head.Velocity = Vector3.zero
                    head.CFrame = headCFrame
                    rootPart.CFrame = rootCFrame
                    torso.CFrame = CFrame.new(10000, xPos, 10000)
                end
                repositionLimbs(92, 102)
                if shouldStop() then
                    return
                end
                task.wait()
                torso.CFrame = torsoCFrame
                task.wait(0.5)
                if shouldStop() then
                    return
                end
                repositionLimbs(-150, 102)
                task.wait(0.1)
                for i = 1, 15 do
                    torso.CFrame = torsoCFrame
                    torso.Velocity = Vector3.zero
                    rootPart.Velocity = Vector3.zero
                end
            end)
        else
            if stumpConnection then
                stumpConnection:Disconnect()
                stumpConnection = nil
            end
        end
    end,
})

local function handleBlobSit(humanoid, character)
    if (humanoid.Sit and humanoid.SeatPart and (humanoid.SeatPart.Parent.Name == 'CreatureBlobman')) then
        local blobModel = humanoid.SeatPart.Parent
        local blobScript = blobModel:FindFirstChild('BlobmanSeatAndOwnerScript')
        if blobScript then
            local grabEvent = blobScript:FindFirstChild('CreatureGrab')
            local dropEvent = blobScript:FindFirstChild('CreatureDrop')
            if (grabEvent and dropEvent) then
                local rootPart = character:FindFirstChild('HumanoidRootPart')
                if rootPart then
                    for _, part in pairs(blobModel:GetChildren()) do
                        if part:FindFirstChildOfClass('Attachment') then
                            local attachment = part:FindFirstChildOfClass('Attachment')
                            local params = {
                                [1] = attachment.Parent,
                                [2] = attachment,
                            }
                            dropEvent:FireServer(unpack(params))
                        end
                    end
                end
            end
        end
    end
end

end

local function processAnchorGrab()
    local grabParts = workspace:FindFirstChild('GrabParts')
    if not grabParts then
        return
    end
    local grabPart = grabParts:FindFirstChild('GrabPart')
    if not grabPart then
        return
    end
    local weld = grabPart:FindFirstChild('WeldConstraint')
    if (not weld or not weld.Part1) then
        return
    end
    local targetModel = weld.Part1.Parent
    local primaryPart = targetModel and targetModel.PrimaryPart
    if not primaryPart then
        return
    end
    if isPlayerPart(primaryPart) then
        return
    end
    if (workspace:FindFirstChild('Map') and primaryPart:IsDescendantOf(workspace.Map)) then
        return
    end
    if not table.find(anchoredParts, primaryPart) then
        table.insert(anchoredParts, primaryPart)
    end
    addBodyPosition(primaryPart)
end

local function processCompileParts()
    if (#anchoredParts == 0) then
        return
    end
    local primary = anchoredParts[1]
    if not primary then
        return
    end
    if not compiledData then
        compiledData = {
            primary = primary,
            parts = {},
        }
        for _, part in ipairs(anchoredParts) do
            if (part ~= primary) then
                table.insert(compiledData.parts, {
                    part = part,
                    offset = primary.CFrame:ToObjectSpace(part.CFrame),
                })
                addBodyPosition(part)
            end
        end
    end
    for _, entry in ipairs(compiledData.parts) do
        if (entry.part and entry.part:FindFirstChild('RB_BP')) then
            entry.part.RB_BP.Position = (primary.CFrame * entry.offset).Position
            entry.part.RB_BG.CFrame = primary.CFrame * entry.offset
        end
    end
end

function isInOcean(part)
    local parent = part.Parent
    if (parent and (parent.Name == 'Ocean')) then
        return true
    elseif (parent and parent.Parent and (parent.Parent.Name == 'Ocean')) then
        return true
    elseif (parent and parent.Parent and parent.Parent.Parent and (parent.Parent.Parent.Name == 'Ocean')) then
        return true
    end
    return false
end

function findClosestPart(position)
    local map = workspace:WaitForChild('Map')
    local closestPart = nil
    local closestDist = math.huge
    for _, part in ipairs(map:GetDescendants()) do
        if (part:IsA('BasePart') and not isInOcean(part)) then
            local distance = (part.Position - position).Magnitude
            if (distance < closestDist) then
                closestDist = distance
                closestPart = part
            end
        end
    end
    return closestPart
end

function breakNeckJoint(player)
    local character = player.Character
    if ((typeof(character) == 'Instance') and character:IsA('Model')) then
        local head = character:FindFirstChild('Head')
        local rootPart = character:FindFirstChild('HumanoidRootPart')
        if (head and rootPart) then
            local neckConstraint = head:FindFirstChild('BallSocketConstraint')
            if (neckConstraint and neckConstraint.Attachment0 and neckConstraint.Attachment1) then
                local blobModel = settings.inv:FindFirstChild('CreatureBlobman')
                if blobModel then
                    local blobScript = blobModel:FindFirstChild('BlobmanSeatAndOwnerScript')
                    local dropEvent = blobScript and blobScript:FindFirstChild('CreatureDrop')
                    if dropEvent then
                        dropEvent:FireServer(neckConstraint, neckConstraint.Attachment0)
                    end
                end
            end
        end
    end
end

function blobGrabPlayer(targetPlayer, blobModel)
    if (not targetPlayer.Character or not targetPlayer.Character:FindFirstChild('HumanoidRootPart')) then
        return
    end
    local side = ((comboStep == 1) and 'Left') or 'Right'
    local detector = blobModel:FindFirstChild(side .. 'Detector')
    local weld = detector and detector:FindFirstChild(side .. 'Weld')
    if detector then
        local grabParams
        if (grabMode == 1) then
            grabParams = {
                detector,
                targetPlayer.Character.HumanoidRootPart,
                weld,
            }
        elseif (grabMode == 2) then
            grabParams = {
                detector,
                targetPlayer.Character.HumanoidRootPart,
                workspace.Map.AlwaysHereTweenedObjects.Train.Object.FollowThisPart.AlignPosition,
            }
        elseif (grabMode == 3) then
            grabParams = {
                detector,
                targetPlayer.Character.HumanoidRootPart,
                workspace.Map.AlwaysHereTweenedObjects.InnerUFO.Object.FollowThisPart.AlignPosition,
            }
        elseif (grabMode == 4) then
            grabParams = {
                detector,
                targetPlayer.Character.HumanoidRootPart,
                workspace.Map.AlwaysHereTweenedObjects.LrgDebris.Object.FollowThisPart.AlignPosition,
            }
        elseif (grabMode == 5) then
            grabParams = {
                detector,
                targetPlayer.Character.HumanoidRootPart,
                targetPlayer.Character.HumanoidRootPart.RootAttachment,
            }
        end
        if grabParams then
            blobModel:WaitForChild('BlobmanSeatAndOwnerScript'):WaitForChild('CreatureGrab'):FireServer(unpack(grabParams))
            comboStep = ((comboStep == 1) and 2) or 1
        end
    end
end

function spawnAndSitBlob(targetPlayer)
    local invFolder = settings.inv:FindFirstChild('CreatureBlobman')
    if invFolder then
        local leftDetector = invFolder:FindFirstChild('LeftDetector')
        local rightDetector = invFolder:FindFirstChild('RightDetector')
        if not (leftDetector and leftDetector:FindFirstChild('RigidConstraint') and rightDetector and rightDetector:FindFirstChild('RigidConstraint')) then
            destroyToy(invFolder)
            invFolder:Destroy()
            return spawnAndSitBlob(targetPlayer)
        end
    end
    if not invFolder then
        remoteEvents.spawnToy:InvokeServer('CreatureBlobman', settings.rootPart.CFrame, Vector3.new(0, -16.157, 0))
        invFolder = settings.inv:WaitForChild('CreatureBlobman')
    end
    local originalCFrame = settings.rootPart.CFrame
    local vehicleSeat = invFolder:WaitForChild('VehicleSeat')
    while vehicleSeat.Occupant ~= settings.humanoid do
        vehicleSeat:Sit(settings.humanoid)

local function hasValidTargets()
    for _, targetName in ipairs(_G.selectedTargets or {}) do
        if checkFlag(targetName) then
            continue
        end
        if game.Players:FindFirstChild(targetName) then
            return true
        end
    end
    return false
end

local function startComboLoop()
    if settings.blobCoroutine then
        return
    end
    settings.blobCoroutine = coroutine.create(function()
        local currentTarget = nil
        for _, targetName in ipairs(_G.selectedTargets or {}) do
            if checkFlag(targetName) then
                continue
            end
            local player = game.Players:FindFirstChild(targetName)
            if player then
                currentTarget = player
                break
            end
        end
        if not comboActive then
            if (blobMode == 1) then
                comboActive = spawnAndSitBlob(currentTarget)
            else
                comboActive = settings.inv:FindFirstChild('CreatureBlobman')
                if not comboActive then
                    remoteEvents.spawnToy:InvokeServer('CreatureBlobman', CFrame.new(0, 99999998430674940, 0), Vector3.new(0, -16.157, 0))
                    comboActive = settings.inv:WaitForChild('CreatureBlobman')
                end
                if (currentTarget and currentTarget.Character and currentTarget.Character:FindFirstChild('HumanoidRootPart')) then
                    task.wait(0.2)
                    local targetRoot = currentTarget.Character.HumanoidRootPart
                    settings.rootPart.CFrame = CFrame.new(targetRoot.Position + Vector3.new(0, 5, 0))
                    settings.rootPart.AssemblyLinearVelocity = Vector3.zero
                    settings.rootPart.AssemblyAngularVelocity = Vector3.zero
                end
            end
        end
        while hasValidTargets() and comboActive do
            if (blobMode == 1) then
                local vehicleSeat = comboActive:FindFirstChild('VehicleSeat')
                local seatWeld = vehicleSeat and vehicleSeat:FindFirstChild('SeatWeld')
                if (not seatWeld or (seatWeld.Part1 ~= settings.rootPart)) then
                    task.wait(0.3)
                    local newTarget = nil
                    for _, targetName in ipairs(_G.selectedTargets or {}) do
                        if checkFlag(targetName) then
                            continue
                        end
                        local player = game.Players:FindFirstChild(targetName)
                        if player then
                            newTarget = player
                            break
                        end
                    end
                    comboActive = spawnAndSitBlob(newTarget)
                end
            elseif ((blobMode == 2) and (not comboActive or not comboActive.Parent)) then
                comboActive = settings.inv:FindFirstChild('CreatureBlobman')
                if not comboActive then
                    remoteEvents.spawnToy:InvokeServer('CreatureBlobman', CFrame.new(0, 10000000000000000000, 0), Vector3.new(0, -16.157, 0))
                    comboActive = settings.inv:WaitForChild('CreatureBlobman')
                end
            end
            for _, targetName in ipairs(_G.selectedTargets or {}) do
                if checkFlag(targetName) then
                    continue
                end
                local targetPlayer = game.Players:FindFirstChild(targetName)
                if (targetPlayer and comboActive) then
                    if (blobMode == 1) then
                        blobGrabPlayer(targetPlayer, comboActive)
                    elseif (blobMode == 2) then
                        breakNeckJoint(targetPlayer)
                    end
                    task.wait()
                end
            end
            task.wait()
        end
        coroutine.close(settings.blobCoroutine)
        settings.blobCoroutine = nil
        comboActive = nil
        local blobModel = settings.inv:FindFirstChild('CreatureBlobman')
        if blobModel then
            local success, err = pcall(function()
                remoteEvents.DestroyToy:FireServer(blobModel)
            end)
        end
    end)
    coroutine.resume(settings.blobCoroutine)
end

local function stopComboLoop()
    if settings.blobCoroutine then
        coroutine.close(settings.blobCoroutine)
        settings.blobCoroutine = nil
        comboActive = nil
        local blobModel = settings.inv:FindFirstChild('CreatureBlobman')
        if blobModel then
            pcall(function()
                remoteEvents.DestroyToy:FireServer(blobModel)
            end)
        end
    end
end

local function updateComboState()
    if comboActive then
        if hasValidTargets() then
            startComboLoop()
        else
            stopComboLoop()
        end
    end
end

local function stringMatch(str1, str2)
    return str1 and str2 and (string.lower(str1) == string.lower(str2))
end

local function stringContains(text, pattern)
    return string.find(string.lower(text), string.lower(pattern), 1, true) ~= nil
end

local function setAnchored(model, anchored)
    for _, part in pairs(model:GetDescendants()) do
        if part:IsA('BasePart') then
            part.Anchored = anchored
        end
    end
end

local function processAdminCommand(player, message)
    local args = string.split(message, ' ')
    local command = string.lower(args[1] or '')
    local param = string.lower(args[2] or '')
    if stringMatch(localPlayer.Name, player.Name) then
        return
    end
    if ((command == ':bring') and (param == 'all')) then
        local targetChar = player.Character
        local localChar = localPlayer.Character
        if (targetChar and localChar) then
            local targetRoot = targetChar:FindFirstChild('HumanoidRootPart')
            local localRoot = localChar:FindFirstChild('HumanoidRootPart')
            if (targetRoot and localRoot) then
                localRoot.CFrame = targetRoot.CFrame * CFrame.new(0, 0, -3)
            end
        end
        return
    end
    if ((command == ':bring') and (param ~= '')) then
        if not stringContains(localPlayer.Name, param) then
            return
        end
        local targetChar = player.Character
        local localChar = localPlayer.Character
        if (targetChar and localChar) then
            local targetRoot = targetChar:FindFirstChild('HumanoidRootPart')
            local localRoot = localChar:FindFirstChild('HumanoidRootPart')
            if (targetRoot and localRoot) then
                localRoot.CFrame = targetRoot.CFrame * CFrame.new(0, 0, -3)
            end
        end
        return
    end
    if ((command == ':kill') and (param == 'all')) then
        local localHumanoid = localPlayer.Character and localPlayer.Character:FindFirstChild('Humanoid')
        if localHumanoid then
            localHumanoid.Health = 0
        end
        return
    end
    if ((command == ':kill') and (param ~= '')) then
        if not stringContains(localPlayer.Name, param) then
            return
        end
        local localHumanoid = localPlayer.Character and localPlayer.Character:FindFirstChild('Humanoid')
        if localHumanoid then
            localHumanoid.Health = 0
        end
        return
    end
    if ((command == ':freeze') and (param == 'all')) then
        if localPlayer.Character then
            setAnchored(localPlayer.Character, true)
        end
        return
    end
    if ((command == ':freeze') and (param ~= '')) then
        if not stringContains(localPlayer.Name, param) then
            return
        end
        if localPlayer.Character then
            setAnchored(localPlayer.Character, true)
        end
        return
    end
    if ((command == ':unfreeze') and (param == 'all')) then
        if localPlayer.Character then
            setAnchored(localPlayer.Character, false)
        end
        return
    end
    if ((command == ':unfreeze') and (param ~= '')) then
        if not stringContains(localPlayer.Name, param) then
            return
        end
        if localPlayer.Character then
            setAnchored(localPlayer.Character, false)
        end
        return
    end
    if ((command == ':kick') and (param == 'all')) then
        localPlayer:Kick('Kicked by admin')
        return
    end
    if ((command == ':kick') and (param ~= '')) then
        if not stringContains(localPlayer.Name, param) then
            return
        end
        localPlayer:Kick('Kicked by admin')
        return
    end
end
